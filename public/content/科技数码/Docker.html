docker version

显示 Docker 客户端和服务端的版本信息。

# 查看版本
docker version

# 仅查看客户端
docker version --format '{{.Client.Version}}'
基础

docker info

显示 Docker 系统范围的信息，包括镜像数、容器数、存储驱动等。

# 系统信息
docker info

# 查看特定信息
docker info --format '{{json .}}' | jq

docker pull

从仓库拉取镜像或仓库到本地。

# 拉取最新版
docker pull ubuntu:latest

# 拉取特定版本
docker pull nginx:1.24

# 从私有仓库拉取
docker pull registry.example.com/myapp:v1
镜像 常用

docker build

通过 Dockerfile 构建镜像。

# 基础构建
docker build -t myapp:latest .

# 指定 Dockerfile
docker build -f Dockerfile.prod -t myapp .

# 不使用缓存
docker build --no-cache -t myapp .

# 构建时变量
docker build --build-arg VERSION=1.0 -t myapp .

docker images

列出本地镜像。

# 列出所有镜像
docker images

# 仅显示 ID
docker images -q

# 显示中间层
docker images -a

# 过滤悬空镜像
docker images --filter "dangling=true"

docker rmi

删除一个或多个镜像。

# 删除指定镜像
docker rmi ubuntu:latest

# 强制删除
docker rmi -f ubuntu

# 删除所有悬空镜像
docker rmi $(docker images -f "dangling=true" -q)

docker tag

给镜像打标签，用于版本控制或推送到不同仓库。

# 标记镜像
docker tag myapp:v1 registry.com/myapp:v1

# 给现有镜像添加 latest 标签
docker tag myapp:1.0 myapp:latest

docker run

创建并启动新容器，这是最常用的命令之一。

# 后台运行并映射端口
docker run -d -p 80:80 --name web nginx

# 交互式终端
docker run -it --rm ubuntu bash

# 挂载卷
docker run -v /host/path:/container/path nginx

# 设置环境变量
docker run -e MYSQL_ROOT_PASSWORD=secret mysql
核心 常用

docker ps

列出容器。

# 运行中的容器
docker ps

# 所有容器（包括停止的）
docker ps -a

# 显示容器大小
docker ps -s

# 过滤
docker ps --filter "status=exited"

docker start/stop/restart

管理容器运行状态。

# 启动容器
docker start container_name

# 停止容器（发送 SIGTERM）
docker stop container_name

# 强制停止（发送 SIGKILL）
docker kill container_name

# 重启
docker restart container_name

docker rm

删除一个或多个容器。

# 删除已停止的容器
docker rm container_id

# 强制删除运行中的容器
docker rm -f container_id

# 删除所有已停止容器
docker container prune

# 删除并挂载卷
docker rm -v container_id

docker exec

在运行中的容器内执行命令，常用于进入容器调试。

# 进入容器交互式 shell
docker exec -it container_name bash

# 以 root 身份执行
docker exec -u 0 container_name sh

# 后台执行命令
docker exec -d container_name touch /tmp/file
调试

docker logs

获取容器日志。

# 查看日志
docker logs container_name

# 实时跟踪（类似 tail -f）
docker logs -f container_name

# 显示最后 100 行
docker logs --tail 100 container_name

# 显示时间戳
docker logs -t container_name

docker cp

在容器和主机之间复制文件。

# 容器到主机
docker cp container:/app/file.txt ./local/

# 主机到容器
docker cp ./local/file.txt container:/app/

# 复制整个目录
docker cp container:/app ./backup

docker inspect

查看容器或镜像的详细信息（JSON 格式）。

# 查看容器信息
docker inspect container_name

# 提取特定字段（如 IP 地址）
docker inspect --format='{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' container

# 查看镜像历史
docker inspect image:tag

docker stats

实时显示容器资源使用情况（CPU、内存、网络、IO）。

# 监控所有容器
docker stats

# 监控特定容器
docker stats container_name

# 不截断输出
docker stats --no-trunc

# 仅显示 ID、CPU、内存
docker stats --format "table {{.Container}}\t{{.CPUPerc}}\t{{.MemUsage}}"

docker network

管理 Docker 网络，包括创建、连接、断开和删除网络。

# 列出网络
docker network ls

# 创建桥接网络
docker network create --driver bridge my-network

# 连接容器到网络
docker network connect my-network container

# 断开网络
docker network disconnect my-network container

# 查看网络详情
docker network inspect my-network
网络

docker volume

管理 Docker 数据卷，用于持久化存储。

# 创建卷
docker volume create my-vol

# 列出卷
docker volume ls

# 查看卷详情
docker volume inspect my-vol

# 删除未使用的卷
docker volume prune

# 删除特定卷
docker volume rm my-vol
存储

docker compose up

构建、（重新）创建、启动和附加到容器（用于 docker-compose.yml）。

# 后台启动所有服务
docker compose up -d

# 构建镜像后启动
docker compose up --build

# 强制重新创建容器
docker compose up --force-recreate

# 缩放服务实例数
docker compose up -d --scale web=3
Compose 编排

docker compose down

停止并移除容器、网络、镜像和卷。

# 停止并删除容器和网络
docker compose down

# 同时删除卷
docker compose down -v

# 同时删除镜像
docker compose down --rmi all

# 删除孤儿容器
docker compose down --remove-orphans

docker compose ps/logs

查看 Compose 项目状态和服务日志。

# 列出运行的服务
docker compose ps

# 查看所有服务日志
docker compose logs -f

# 查看特定服务日志
docker compose logs -f web

# 执行服务命令
docker compose exec web sh

docker system

管理 Docker 系统，包括清理未使用的数据。

# 清理未使用的数据（容器、网络、镜像、缓存）
docker system prune

# 清理所有未使用数据（包括卷）
docker system prune -a --volumes

# 查看磁盘使用
docker system df

# 详细查看
docker system df -v
维护 危险

docker top

查看容器中运行的进程（类似 Linux top）。

# 查看容器进程
docker top container_name

# 显示帮助
docker --help

# 查看事件
docker events --since '1h'

docker push

推送镜像到仓库。

# 推送到 Docker Hub
docker push username/image:tag

# 推送到私有仓库
docker push registry.example.com/project/image:v1

# 登录仓库
docker login registry.example.com

# 登出
docker logout
仓库

docker commit

从容器的更改创建新镜像（不推荐用于生产，建议使用 Dockerfile）。

# 提交容器为新镜像
docker commit container_id new_image:v2

# 带作者和消息
docker commit -a "Author" -m "message" container image:tag
高级

docker export/import

导出容器文件系统为 tar 包，或从 tar 包创建镜像。

# 导出容器
docker export container_id > backup.tar

# 导入为镜像
docker import backup.tar new_image:tag

# 保存镜像（带历史）
docker save image:tag > image.tar

# 加载镜像
docker load < image.tar

docker buildx

Docker BuildKit 扩展，支持多平台构建和高级构建功能。

# 多平台构建
docker buildx build --platform linux/amd64,linux/arm64 -t myimage .

# 构建并推送
docker buildx build --push -t user/image .

# 创建 builder 实例
docker buildx create --name mybuilder --use
